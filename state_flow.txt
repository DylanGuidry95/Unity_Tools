# Main Application Flow

[AutoTransition = false]

[Parameters]
{
    (float)Life;
    (string)name;
    (int)test;
    (bool)IsDead
}

[State]
{
    Initialize;
    Launcher;
    Introduction;
    Scenario;
    Debrief;
    Score;
    SubmitData
}

[Transitions]
{
    Initialize>Introduction;
    Initialize>Launcher;
    Launcher>Introduction;
    Introduction>Scenario;
    Introduction>Launcher;
    Scenario>Introduction;
    Scenario>Debrief;
    Debrief>Score;
    Score>SubmitData;
    Score>Introduction;
    Score>Launcher;
    SubmitData>Launcher;
    SubmitData>Introduction
}

[Condition = Initialize>Introduction]
{
    Life > 1.3;
    test < 5;
    name == Dylan;
    IsDead != true
}
